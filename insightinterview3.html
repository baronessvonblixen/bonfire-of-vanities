import pandas as pd
import numpy as np
import matplotlib
import matplotlib.pyplot as plt
from scipy.stats import normaltest

#load txt file as a dataframe. make wrong entries NaNs
data = pd.read_csv('Desktop/breast-cancer-wisconsin.txt', index_col = "Index", na_values={'Class':['No idea', '#','40','20', '?']})

#clean data of duplicate and invalid entries
dataclean = data.drop_duplicates()
dataclean_filter = dataclean.dropna(axis=0, how='any', thresh=None, subset=None, inplace=False)
#print(dataclean_filter)
dataclean_filter.info()

#slice data in half to create a training data set and test data set
shape = dataclean_filter.shape
#print(shape)
length = shape[0]-1
#print(length)
halflength = length/2
#print(halflength)
training = dataclean_filter.head(344)
#print(training)
training.info()

#make sure there are only one of each biopsy ID in the dataset
close_id = training['ID'].values
frequency_id = np.bincount(close_id)
plt.plot(frequency_id)
plt.xlim(0,1.5e6)
plt.xlabel('ID number')
plt.ylabel('Number in Data Set')
plt.title('Frequency of Duplicate IDs in Training Set')
plt.show()

#There are many many duplicate data sets in the work.  Possibilities: 1. same patient different biopsies; 
#2.Same biopsy different pathologists; 3. accidental redundacy, etc. Histogram of training set revealed same ID 
#with different scores. This removes option 3. for these data points. Still, in an ideal world, should not treat 
#these as independent. For time being, remove total duplicates, keep different scores.  If anything, it will 
#lower the fidelity of the model. 

#data type for all catagories not the same (integers vs. non-null objects) Because I am planning on using numpy 
#array-like structures for analysis, shouldn't be a problem. But keep in mind with debugging.
